_models = {}

def load_all():
    _models['linear']  = joblib.load(config.LINEAR_MODEL_PATH)
    _models['dt']      = joblib.load(config.DT_MODEL_PATH)
    _models['knn']     = joblib.load(config.KNN_MODEL_PATH)
    _models['kmeans']  = joblib.load(config.KMEANS_MODEL_PATH)

def get_linear():        return _models['linear']
def get_decision_tree(): return _models['dt']
def get_knn():           return _models['knn']
def get_kmeans():        return _models['kmeans']
